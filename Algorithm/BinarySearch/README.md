func binarySearch(_ array: [Int], target: Int) -> Bool {
    var start = 0
    var end = array.count - 1
    
    while start <= end {
        let mid = (start + end) / 2
        
        if array[mid] == target { return true }
        if array[mid] > target {
            end = mid - 1
        } else {
            start = mid + 1
        }
    }
    return false
}
